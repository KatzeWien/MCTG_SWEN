CREATE TABLE IF NOT EXISTS users (
username VARCHAR(100) PRIMARY KEY,
password VARCHAR(100),
coins INT,
elo INT,
wins INT,
losses INT,
bio VARCHAR(250) DEFAULT 'n/a',
image VARCHAR(250) DEFAULT 'n/a', 
name VARCHAR(100) DEFAULT 'n/a'
);

CREATE TABLE IF NOT EXISTS cards (
id VARCHAR(250) PRIMARY KEY,
name VARCHAR(250),
damage DOUBLE PRECISION,
element VARCHAR(50),
cardtype VARCHAR(50)
);

CREATE TABLE IF NOT EXISTS packages (
id SERIAL PRIMARY KEY,
firstcard VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE,
secondcard VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE,
thirdcard VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE,
forthcard VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE,
fifthcard VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS stacks (
id SERIAL PRIMARY KEY,
userid VARCHAR(100) REFERENCES users(username) ON DELETE CASCADE,
cardid VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS decks (
id SERIAL PRIMARY KEY,
userid VARCHAR(100) REFERENCES users(username) ON DELETE CASCADE,
cardid VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS trades (
id VARCHAR(250) PRIMARY KEY,
userid VARCHAR(100) REFERENCES users(username) ON DELETE CASCADE,
cardid VARCHAR(250) REFERENCES cards(id) ON DELETE CASCADE,
type VARCHAR(50),
damage DOUBLE PRECISION
);
